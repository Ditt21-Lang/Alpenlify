int get_currently_player_music_length(PlayerHandle *handle) {
    float outLen;
    ma_result res;
    res = ma_sound_get_length_in_seconds(&(handle->sound), &outLen);

    if (res != MA_SUCCESS) {
        printf("ERROR: %s", ma_result_description(res));
        return -1;
    }
    
    return (int)floorf(outLen);
}
